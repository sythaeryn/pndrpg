FILE(GLOB SRC *.cpp *.h)

ADD_EXECUTABLE(build_interface ${SRC})

INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(build_interface ${LIBXML2_LIBRARIES} ${PLATFORM_LINKFLAGS} nelmisc)
IF(WIN32)
  SET_TARGET_PROPERTIES(build_interface PROPERTIES 
    LINK_FLAGS_DEBUG "${CMAKE_LINK_FLAGS_DEBUG}"
    LINK_FLAGS_RELEASE "${CMAKE_LINK_FLAGS_RELEASE}"
    PROJECT_LABEL "Tools, 3D: build_interface")
ENDIF(WIN32)
ADD_DEFINITIONS(${LIBXML2_DEFINITIONS})

INSTALL(TARGETS build_interface RUNTIME DESTINATION bin COMPONENT tools3d)
